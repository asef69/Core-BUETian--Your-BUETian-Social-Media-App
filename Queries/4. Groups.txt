2.Get Group Details

CREATE INDEX IF NOT EXISTS idx_group_members_user_gid ON group_members(user_id);
CREATE INDEX IF NOT EXISTS idx_group_members_status_gid ON group_members(status);


SELECT 
    g.id as group_id,
    g.name,
    g.description,
    g.admin_id,
    u.name as admin_name,
    g.is_private,
    g.cover_image,
    COUNT(DISTINCT gm_all.user_id) FILTER (WHERE gm_all.status = 'accepted') AS members_count,
    COUNT(DISTINCT p.id) AS posts_count,
    CASE WHEN p_user_id IS NOT NULL THEN 
        MAX(CASE WHEN gm_user.user_id IS NOT NULL AND gm_user.status = 'accepted' THEN TRUE ELSE FALSE END)
    ELSE FALSE END AS is_member,
    MAX(gm_user.status) AS member_status,
    g.created_at
FROM groups g
INNER JOIN users u ON g.admin_id = u.id
LEFT JOIN group_members gm_all ON g.id = gm_all.group_id 
LEFT JOIN posts p ON g.id = p.group_id                  
LEFT JOIN group_members gm_user ON g.id = gm_user.group_id AND gm_user.user_id = p_user_id  -- current user membership
WHERE g.id = p_group_id
GROUP BY g.id, g.name, g.description, g.admin_id, u.name, g.is_private, g.cover_image, g.created_at;
